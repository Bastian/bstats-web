import com.google.inject.Inject;
import org.spongepowered.api.event.Listener;
import org.spongepowered.api.event.game.state.GameStartedServerEvent;
import org.spongepowered.api.plugin.Plugin;

@Plugin(id = "exampleplugin", name = "ExamplePlugin", version = "1.0")
public class ExamplePlugin {

    private final Metrics metrics;

    // The metricsFactory parameter gets injected using @Inject
    // See https://docs.spongepowered.org/stable/en/plugin/injection.html
    @Inject
    public ExamplePlugin(Metrics.Factory metricsFactory) {
        // You can find the plugin id of your plugins on
        // the page https://bstats.org/what-is-my-plugin-id
        int pluginId = {{pluginId}};
        metrics = metricsFactory.make(pluginId);
    }

    // Optional: Add custom charts
    @Listener
    public void onServerStart(GameStartedServerEvent event) {
        metrics.addCustomChart(
            new Metrics.SimplePie("chart_id", () -> "My value")
        );
    }

}
